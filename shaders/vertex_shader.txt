# version 330 core
layout(location = 0) in vec3 in_vertex;
layout(location = 1) in vec2 in_texCoords;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

out vec2 texCoords;
out vec3 position;

void main() {
	gl_Position = projection * view * model * vec4(in_vertex, 1.0);
	texCoords = in_texCoords;
	position = in_vertex;
}
